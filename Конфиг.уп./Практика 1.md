#Задание 3

```
#!/bin/bash
if [ -z "$1" ]; then
  echo "Использование: $0 \"Текст для баннера\""
  exit 1
fi
text="$1"
length=${#text}
banner_length=$((length + 4))
printf '+%s+\n' "$(printf '%*s' "$banner_length" '' | tr ' ' '-')"
printf '| %s |\n' "$text"
printf '+%s+\n' "$(printf '%*s' "$banner_length" '' | tr ' ' '-')"
```
